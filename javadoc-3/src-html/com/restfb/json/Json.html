022package com.restfb.json;
023
024import java.io.IOException;
025import java.io.Reader;
026import java.util.Objects;
027
028/**
029 * This class serves as the entry point to the minimal-json API.
030 * <p>
031 * To <strong>parse</strong> a given JSON input, use the <code>parse()</code> methods like in this example:
032 * </p>
033 * 
034 * <pre>
035 * JsonObject object = Json.parse(string).asObject();
036 * </pre>
037 * <p>
038 * To <strong>create</strong> a JSON data structure to be serialized, use the methods <code>value()</code>,
039 * <code>array()</code>, and <code>object()</code>. For example, the following snippet will produce the JSON string
040 * <em>{"foo": 23, "bar": true}</em>:
041 * </p>
042 * 
043 * <pre>
044 * String string = Json.object().add("foo", 23).add("bar", true).toString();
045 * </pre>
046 * <p>
047 * To create a JSON array from a given Java array, you can use one of the <code>array()</code> methods with varargs
048 * parameters:
049 * </p>
050 * 
051 * <pre>
052 * String[] names = ...
053 * JsonArray array = Json.array(names);
054 * </pre>
055 */
056public final class Json {
057
058  // String constants
059  private static final String VALUES_IS_NULL = "values is null";
060  private static final String STRING_IS_NULL = "string is null";
061  private static final String READER_IS_NULL = "reader is null";
062  private static final String INFINITE_AND_NAN = "Infinite and NaN values not permitted in JSON";
063
064  private Json() {
065    // not meant to be instantiated
066  }
067
068  /**
069   * Represents the JSON literal <code>null</code>.
070   */
071  public static final JsonValue NULL = new JsonLiteral("null");
072
073  /**
074   * Represents the JSON literal <code>true</code>.
075   */
076  public static final JsonValue TRUE = new JsonLiteral("true");
077
078  /**
079   * Represents the JSON literal <code>false</code>.
080   */
081  public static final JsonValue FALSE = new JsonLiteral("false");
082
083  /**
084   * Returns a JsonValue instance that represents the given <code>int</code> value.
085   *
086   * @param value
087   *          the value to get a JSON representation for
088   * @return a JSON value that represents the given value
089   */
090  public static JsonValue value(int value) {
091    return new JsonNumber(Integer.toString(value, 10));
092  }
093
094  /**
095   * Returns a JsonValue instance that represents the given <code>long</code> value.
096   *
097   * @param value
098   *          the value to get a JSON representation for
099   * @return a JSON value that represents the given value
100   */
101  public static JsonValue value(long value) {
102    return new JsonNumber(Long.toString(value, 10));
103  }
104
105  /**
106   * Returns a JsonValue instance that represents the given <code>float</code> value.
107   *
108   * @param value
109   *          the value to get a JSON representation for
110   * @return a JSON value that represents the given value
111   */
112  public static JsonValue value(float value) {
113    if (Float.isInfinite(value) || Float.isNaN(value)) {
114      throw new IllegalArgumentException(INFINITE_AND_NAN);
115    }
116    return new JsonNumber(cutOffPointZero(Float.toString(value)));
117  }
118
119  /**
120   * Returns a JsonValue instance that represents the given <code>double</code> value.
121   *
122   * @param value
123   *          the value to get a JSON representation for
124   * @return a JSON value that represents the given value
125   */
126  public static JsonValue value(double value) {
127    if (Double.isInfinite(value) || Double.isNaN(value)) {
128      throw new IllegalArgumentException(INFINITE_AND_NAN);
129    }
130    return new JsonNumber(cutOffPointZero(Double.toString(value)));
131  }
132
133  /**
134   * Returns a JsonValue instance that represents the given string.
135   *
136   * @param string
137   *          the string to get a JSON representation for
138   * @return a JSON value that represents the given string
139   */
140  public static JsonValue value(String string) {
141    return string == null ? NULL : new JsonString(string);
142  }
143
144  /**
145   * Returns a JsonValue instance that represents the given <code>boolean</code> value.
146   *
147   * @param value
148   *          the value to get a JSON representation for
149   * @return a JSON value that represents the given value
150   */
151  public static JsonValue value(boolean value) {
152    return value ? TRUE : FALSE;
153  }
154
155  /**
156   * Creates a new empty JsonArray. This is equivalent to creating a new JsonArray using the constructor.
157   *
158   * @return a new empty JSON array
159   */
160  public static JsonArray array() {
161    return new JsonArray();
162  }
163
164  /**
165   * Creates a new JsonArray that contains the JSON representations of the given <code>int</code> values.
166   *
167   * @param values
168   *          the values to be included in the new JSON array
169   * @return a new JSON array that contains the given values
170   */
171  public static JsonArray array(int... values) {
172    if (values == null) {
173      throw new NullPointerException(VALUES_IS_NULL);
174    }
175    JsonArray array = new JsonArray();
176    for (int value : values) {
177      array.add(value);
178    }
179    return array;
180  }
181
182  /**
183   * Creates a new JsonArray that contains the JSON representations of the given <code>long</code> values.
184   *
185   * @param values
186   *          the values to be included in the new JSON array
187   * @return a new JSON array that contains the given values
188   */
189  public static JsonArray array(long... values) {
190    Objects.requireNonNull(values, VALUES_IS_NULL);
191    JsonArray array = new JsonArray();
192    for (long value : values) {
193      array.add(value);
194    }
195    return array;
196  }
197
198  /**
199   * Creates a new JsonArray that contains the JSON representations of the given <code>float</code> values.
200   *
201   * @param values
202   *          the values to be included in the new JSON array
203   * @return a new JSON array that contains the given values
204   */
205  public static JsonArray array(float... values) {
206    Objects.requireNonNull(values, VALUES_IS_NULL);
207    JsonArray array = new JsonArray();
208    for (float value : values) {
209      array.add(value);
210    }
211    return array;
212  }

215   * Creates a new JsonArray that contains the JSON representations of the given <code>double</code> values.
216   *
217   * @param values
218   *          the values to be included in the new JSON array
219   * @return a new JSON array that contains the given values
220   */
221  public static JsonArray array(double... values) {
222    Objects.requireNonNull(values, VALUES_IS_NULL);
223    JsonArray array = new JsonArray();

225      array.add(value);
226    }
227    return array;
228  }

231   * Creates a new JsonArray that contains the JSON representations of the given <code>boolean</code> values.
232   *
233   * @param values
234   *          the values to be included in the new JSON array
235   * @return a new JSON array that contains the given values
236   */
237  public static JsonArray array(boolean... values) {
238    Objects.requireNonNull(values, VALUES_IS_NULL);
239    JsonArray array = new JsonArray();
240    for (boolean value : values) {
241      array.add(value);
242    }
243    return array;
244  }

246  /**
247   * Creates a new JsonArray that contains the JSON representations of the given strings.
248   *
249   * @param strings
250   *          the strings to be included in the new JSON array
251   * @return a new JSON array that contains the given strings
252   */
253  public static JsonArray array(String... strings) {
254    Objects.requireNonNull(strings, VALUES_IS_NULL);
255    JsonArray array = new JsonArray();

257      array.add(value);

259    return array;

261
262  /**
263   * Creates a new empty JsonObject. This is equivalent to creating a new JsonObject using the constructor.
264   *
265   * @return a new empty JSON object
266   */
267  public static JsonObject object() {
268    return new JsonObject();
269  }
270
271  /**
272   * Parses the given input string as JSON. The input must contain a valid JSON value, optionally padded with
273   * whitespace.
274   *
275   * @param string
276   *          the input string, must be valid JSON
277   * @return a value that represents the parsed JSON
278   * @throws ParseException
279   *           if the input is not valid JSON
280   */
281  public static JsonValue parse(String string) {
282    Objects.requireNonNull(string, STRING_IS_NULL);
283    DefaultHandler handler = new DefaultHandler();
284    new JsonParser(handler).parse(string);
285    return handler.getValue();
286  }
287
288  /**
289   * Reads the entire input from the given reader and parses it as JSON. The input must contain a
290   * valid JSON value, optionally padded with whitespace.
291   * <p>
292   * Characters are read in chunks into an input buffer. Hence, wrapping a reader in an additional
293   * <code>BufferedReader</code> likely won't improve reading performance.
294   * </p>
295   *
296   * @param reader
297   *          the reader to read the JSON value from
298   * @return a value that represents the parsed JSON
299   * @throws IOException
300   *           if an I/O error occurs in the reader
301   * @throws ParseException
302   *           if the input is not valid JSON
303   */
304  public static JsonValue parse(Reader reader) throws IOException {
305    Objects.requireNonNull(reader, READER_IS_NULL);
306    DefaultHandler handler = new DefaultHandler();
307    new JsonParser(handler).parse(reader);
308    return handler.getValue();
309  }
310
311  private static String cutOffPointZero(String string) {
312    if (string.endsWith(".0")) {
313      return string.substring(0, string.length() - 2);
314    }
315    return string;
316  }
317
318  static class DefaultHandler extends JsonHandler<JsonArray, JsonObject> {
319
320    protected JsonValue value;
321
322    @Override
323    public JsonArray startArray() {
324      return new JsonArray();
325    }
326
327    @Override
328    public JsonObject startObject() {
329      return new JsonObject();
330    }
331
332    @Override
333    public void endNull() {
334      value = NULL;
335    }
336
337    @Override
338    public void endBoolean(boolean bool) {
339      value = bool ? TRUE : FALSE;
340    }
341
342    @Override
343    public void endString(String string) {
344      value = new JsonString(string);
345    }
346
347    @Override
348    public void endNumber(String string) {
349      value = new JsonNumber(string);
350    }
351
352    @Override
353    public void endArray(JsonArray array) {
354      value = array;
355    }
356
357    @Override
358    public void endObject(JsonObject object) {
359      value = object;
360    }
361
362    @Override
363    public void endArrayValue(JsonArray array) {
364      array.add(value);
365    }
366
367    @Override
368    public void endObjectValue(JsonObject object, String name) {
369      object.add(name, value);
370    }
371
372    JsonValue getValue() {
373      return value;
374    }
375
376  }
