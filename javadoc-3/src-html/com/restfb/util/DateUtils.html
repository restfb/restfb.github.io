022package com.restfb.util;
023
024import static com.restfb.logging.RestFBLogger.UTILS_LOGGER;
025
026import java.text.ParseException;
027import java.util.Date;
028import java.util.Optional;
029
030/**
031 * A collection of date-handling utility methods.
032 * 
033 * @author <a href="http://restfb.com">Mark Allen</a>
034 * @since 1.6
035 */
036public final class DateUtils {
037  /**
038   * Facebook "long" date format (IETF RFC 3339). Example: {@code 2010-02-28T16:11:08+0000}
039   */
040  public static final String FACEBOOK_LONG_DATE_FORMAT = "yyyy-MM-dd'T'HH:mm:ssZ";
041
042  /**
043   * Facebook "long" date format (IETF RFC 3339) without a timezone component. Example: {@code 2010-02-28T16:11:08}
044   */
045  public static final String FACEBOOK_LONG_DATE_FORMAT_WITHOUT_TIMEZONE = "yyyy-MM-dd'T'HH:mm:ss";
046
047  /**
048   * Facebook "long" date format (IETF RFC 3339) without a timezone or seconds component. Example:
049   * {@code 2010-02-28T16:11}
050   */
051  public static final String FACEBOOK_LONG_DATE_FORMAT_WITHOUT_TIMEZONE_OR_SECONDS = "yyyy-MM-dd'T'HH:mm";
052
053  /**
054   * Facebook short date format. Example: {@code 04/15/1984}
055   */
056  public static final String FACEBOOK_SHORT_DATE_FORMAT = "MM/dd/yyyy";
057
058  /**
059   * Facebook alternate short date format. Example: {@code 2012-09-15}
060   */
061  public static final String FACEBOOK_ALTERNATE_SHORT_DATE_FORMAT = "yyyy-MM-dd";
062
063  /**
064   * Facebook month-year only date format. Example: {@code Example: 2007-03}
065   */
066  public static final String FACEBOOK_MONTH_YEAR_DATE_FORMAT = "yyyy-MM";
067
068  /**
069   * DateFormatStrategy (default: SimpleDateFormat).
070   */
071  private static DateFormatStrategy strategy = new SimpleDateFormatStrategy();
072
073  /**
074   * Prevents instantiation.
075   */
076  private DateUtils() {
077    // Prevents instantiation
078  }
079
080  /**
081   * Returns a Java representation of a Facebook "long" {@code date} string, or the number of seconds since the epoch.
082   * <p>
083   * Supports dates with or without timezone information.
084   * 
085   * @param date
086   *          Facebook {@code date} string.
087   * @return Java date representation of the given Facebook "long" {@code date} string or {@code null} if {@code date}
088   *         is {@code null} or invalid.
089   */
090  public static Date toDateFromLongFormat(String date) {
091    if (date == null) {
092      return null;
093    }
094
095    // Is this an all-digit date? Then assume it's the "seconds since epoch"
096    // variant
097    if (date.trim().matches("\\d+")) {
098      return new Date(Long.parseLong(date) * 1000L);
099    }
100
101    Date parsedDate = toDateWithFormatString(date, FACEBOOK_LONG_DATE_FORMAT);
102
103    // Fall back to variant without timezone if the initial parse fails
104    if (parsedDate == null) {
105      parsedDate = toDateWithFormatString(date, FACEBOOK_LONG_DATE_FORMAT_WITHOUT_TIMEZONE);
106    }
107
108    // Fall back to variant without seconds if secondary parse fails
109    if (parsedDate == null) {
110      parsedDate = toDateWithFormatString(date, FACEBOOK_LONG_DATE_FORMAT_WITHOUT_TIMEZONE_OR_SECONDS);
111    }
112
113    return parsedDate;
114  }
115
116  /**
117   * Returns a Java representation of a Facebook "short" {@code date} string.
118   * 
119   * @param date
120   *          Facebook {@code date} string.
121   * @return Java date representation of the given Facebook "short" {@code date} string or {@code null} if {@code date}
122   *         is {@code null} or invalid.
123   */
124  public static Date toDateFromShortFormat(String date) {
125    if (date == null) {
126      return null;
127    }
128
129    Date parsedDate = toDateWithFormatString(date, FACEBOOK_SHORT_DATE_FORMAT);
130
131    // Fall back to variant if initial parse fails
132    if (parsedDate == null) {
133      parsedDate = toDateWithFormatString(date, FACEBOOK_ALTERNATE_SHORT_DATE_FORMAT);
134    }
135
136    return parsedDate;
137  }
138
139  /**
140   * Returns a Java representation of a Facebook "month-year" {@code date} string.
141   * 
142   * @param date
143   *          Facebook {@code date} string.
144   * @return Java date representation of the given Facebook "month-year" {@code date} string or {@code null} if
145   *         {@code date} is {@code null} or invalid.
146   */
147  public static Date toDateFromMonthYearFormat(String date) {
148    if (date == null) {
149      return null;
150    }
151
152    if ("0000-00".equals(date)) {
153      return null;
154    }
155
156    return toDateWithFormatString(date, FACEBOOK_MONTH_YEAR_DATE_FORMAT);
157  }
158
159  /**
160   * Returns a String representation of a {@code date} object
161   * 
162   * @param date
163   *          as Date
164   * @return String representation of a {@code date} object. The String is in the form {@code 2010-02-28T16:11:08}
165   */
166  public static String toLongFormatFromDate(Date date) {
167    return Optional.ofNullable(date).map(strategy.formatFor(FACEBOOK_LONG_DATE_FORMAT_WITHOUT_TIMEZONE)::format).orElse(null);
168  }
169
170  /**
171   * Returns a <strong>short</strong> String representation of a {@code date} object
172   *
173   * @param date
174   *          as Date
175   * @return String representation of a {@code date} object. The String is in the form {@code 2019-06-14}
176   */
177  public static String toShortFormatFromDate(Date date) {
178    return Optional.ofNullable(date).map(strategy.formatFor(FACEBOOK_ALTERNATE_SHORT_DATE_FORMAT)::format).orElse(null);
179  }
180
181  /**
182   * Returns a Java representation of a {@code date} string.
183   * 
184   * @param date
185   *          Date in string format.
186   * @return Java date representation of the given {@code date} string or {@code null} if {@code date} is {@code null}
187   *         or invalid.
188   */
189  private static Date toDateWithFormatString(String date, String format) {
190    if (date == null) {
191      return null;
192    }
193
194    try {
195      return strategy.formatFor(format).parse(date);
196    } catch (ParseException e) {
197      UTILS_LOGGER.trace("Unable to parse date '{}' using format string '{}': {}", date, format, e);
198
199      return null;
200    }
201  }
202
203  /**
204   * get the current DateFormatStrategy.
205   * 
206   * @return the current DateFormatStrategy
207   */
208  public static DateFormatStrategy getDateFormatStrategy() {
209    return strategy;
210  }
211
212  /**
213   * set the {@link DateFormatStrategy}.
214   * 
215   * default value: {@link SimpleDateFormatStrategy}
216   * 
217   * @param dateFormatStrategy
218   *          the used {@link DateFormatStrategy}
219   * 
220   */
221  public static void setDateFormatStrategy(DateFormatStrategy dateFormatStrategy) {
222    strategy = dateFormatStrategy;
223  }
224}
