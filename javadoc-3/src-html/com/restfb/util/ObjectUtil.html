022package com.restfb.util;
023
024import static com.restfb.util.StringUtils.isBlank;
025
026import java.util.Collection;
027import java.util.Map;
028import java.util.Objects;
029
030public class ObjectUtil {
031
032  private ObjectUtil() {
033    // prevent instantiation
034  }
035
036  /**
037   * Ensures that {@code obj} isn't {@code null} or an empty string.
038   *
039   * @param obj
040   *          The parameter to check.
041   * @param errorText
042   *          The exception message.
043   * @throws IllegalArgumentException
044   *           If {@code obj} is {@code null} or an empty string.
045   */
046  public static String requireNotEmpty(String obj, String errorText) {
047    if (isBlank(obj)) {
048      throw new IllegalArgumentException(errorText);
049    }
050    return obj;
051  }
052
053  /**
054   * Checks is the object is a empty 'collection' or 'map'.
055   * 
056   * @param obj
057   *          the object that is checked
058   * @return {@code true} if the given object is a empty collection or an empty map, {@code false} otherwise
059   */
060  public static boolean isEmptyCollectionOrMap(Object obj) {
061    if (obj instanceof Collection) {
062      return ((Collection) obj).isEmpty();
063    }
064
065    return (obj instanceof Map && ((Map) obj).isEmpty());
066  }
067
068  /**
069   * Ensures that {@code parameter} isn't {@code null} or an empty string.
070   *
071   * @param parameterName
072   *          The name of the parameter (to be used in exception message).
073   * @param parameter
074   *          The parameter to check.
075   * @throws IllegalArgumentException
076   *           If {@code parameter} is {@code null} or an empty string.
077   */
078  public static void verifyParameterPresence(String parameterName, String parameter) {
079    verifyParameterPresence(parameterName, (Object) parameter);
080    requireNotEmpty(parameter, "The '" + parameterName + "' parameter cannot be an empty string.");
081  }
082
083  /**
084   * Ensures that {@code parameter} isn't {@code null}.
085   *
086   * @param parameterName
087   *          The name of the parameter (to be used in exception message).
088   * @param parameter
089   *          The parameter to check.
090   * @throws NullPointerException
091   *           If {@code parameter} is {@code null}.
092   */
093  public static void verifyParameterPresence(String parameterName, Object parameter) {
094    Objects.requireNonNull(parameter, "The '" + parameterName + "' parameter cannot be null.");
095  }
096}
