213  @Override
214  public boolean deleteObject(String object, Parameter... parameters) {
215    verifyParameterPresence("object", object);
216
217    String responseString = makeRequest(object, true, true, null, parameters);
218
219    try {
220      JsonValue jObj = Json.parse(responseString);
221      boolean success = false;
222      if (jObj.isObject()) {
223        if (jObj.asObject().get("success") != null) {
224          success = jObj.asObject().get("success").asBoolean();
225        }
226        if (jObj.asObject().get("result") != null) {
227          success = jObj.asObject().get("result").asString().contains("Successfully deleted");
228        }
229      } else {
230        success = jObj.asBoolean();
231      }
232      return success;
233    } catch (ParseException jex) {
234      CLIENT_LOGGER.trace("no valid JSON returned while deleting a object, using returned String instead", jex);
235      return "true".equals(responseString);
236    }
237  }

242  @Override
243  public <T> Connection<T> fetchConnection(String connection, Class<T> connectionType, Parameter... parameters) {
244    verifyParameterPresence("connection", connection);
245    verifyParameterPresence("connectionType", connectionType);
246    Connection<T> conn = new Connection<>(this, makeRequest(connection, parameters), connectionType);
247
248    if (conn.getNextPageUrl() == null && conn.getAfterCursor() != null) {
249      String fullUrl = createEndpointForApiCall(connection, false);
250      String paramString = toParameterString(parameters);
251      fullUrl = UrlUtils.replaceOrAddQueryParameter(fullUrl + "?" + paramString, "after", conn.getAfterCursor());
252      conn.setNextPageUrl(fullUrl);
253    }
254
255    if (conn.getPreviousPageUrl() == null && conn.getBeforeCursor() != null) {
256      String fullUrl = createEndpointForApiCall(connection, false);
257      String paramString = toParameterString(parameters);
258      fullUrl = UrlUtils.replaceOrAddQueryParameter(fullUrl + "?" + paramString, "before", conn.getBeforeCursor());
259      conn.setPreviousPageUrl(fullUrl);
260    }
261
262    return conn;
263  }

266   * @see com.restfb.FacebookClient#fetchConnectionPage(java.lang.String, java.lang.Class)
267   */
268  @Override
269  public <T> Connection<T> fetchConnectionPage(final String connectionPageUrl, Class<T> connectionType) {
270    String connectionJson;
271    if (!isBlank(accessToken) && !isBlank(appSecret)) {
272      connectionJson = makeRequestAndProcessResponse(new Requestor() {
273        @Override
274        public Response makeRequest() throws IOException {
275          return webRequestor.executeGet(String.format("%s&%s=%s", connectionPageUrl,
276            urlEncode(APP_SECRET_PROOF_PARAM_NAME), obtainAppSecretProof(accessToken, appSecret)));
277        }
278      });
279    } else {
280      connectionJson = makeRequestAndProcessResponse(new Requestor() {
281        @Override
282        public Response makeRequest() throws IOException {
283          return webRequestor.executeGet(connectionPageUrl);
284        }
285      });
286    }
287
288    Connection<T> conn = new Connection<>(this, connectionJson, connectionType);
289
290    if (conn.getNextPageUrl() == null && conn.getAfterCursor() != null) {
291      String fullUrl = UrlUtils.removeQueryParameter(connectionPageUrl, "before");
292      fullUrl = UrlUtils.replaceOrAddQueryParameter(fullUrl, "after", conn.getAfterCursor());
293      conn.setNextPageUrl(fullUrl);
294    }
295
296    if (conn.getPreviousPageUrl() == null && conn.getBeforeCursor() != null) {
297      String fullUrl = UrlUtils.removeQueryParameter(connectionPageUrl, "after");
298      fullUrl = UrlUtils.replaceOrAddQueryParameter(fullUrl, "before", conn.getBeforeCursor());
299      conn.setPreviousPageUrl(fullUrl);
300    }
301
302    return conn;
303  }

315  /**
316   * @see com.restfb.FacebookClient#fetchObjects(java.util.List, java.lang.Class, com.restfb.Parameter[])
317   */
318  @Override
319  @SuppressWarnings("unchecked")
320  public <T> T fetchObjects(List<String> ids, Class<T> objectType, Parameter... parameters) {
321    verifyParameterPresence("ids", ids);
322    verifyParameterPresence("connectionType", objectType);
323
324    if (ids.isEmpty()) {
325      throw new IllegalArgumentException("The list of IDs cannot be empty.");
326    }
327
328    for (Parameter parameter : parameters) {
329      if (IDS_PARAM_NAME.equals(parameter.name)) {
330        throw new IllegalArgumentException("You cannot specify the '" + IDS_PARAM_NAME + "' URL parameter yourself - "
331            + "RestFB will populate this for you with " + "the list of IDs you passed to this method.");
332      }
333    }
334
335    JsonArray idArray = new JsonArray();
336    // Normalize the IDs
337    for (String id : ids) {
338      if (StringUtils.isBlank(id)) {
339        throw new IllegalArgumentException("The list of IDs cannot contain blank strings.");
340      }
341      idArray.add(id.trim());
342    }
343
344    try {
345      String jsonString = makeRequest("",
346        parametersWithAdditionalParameter(Parameter.with(IDS_PARAM_NAME, idArray.toString()), parameters));
347
348      return jsonMapper.toJavaObject(jsonString, objectType);
349    } catch (ParseException e) {
350      throw new FacebookJsonMappingException("Unable to map connection JSON to Java objects", e);
351    }
352  }

571  @Override
572  @SuppressWarnings("unchecked")
573  public <T> T parseSignedRequest(String signedRequest, String appSecret, Class<T> objectType) {
574    verifyParameterPresence("signedRequest", signedRequest);
575    verifyParameterPresence("appSecret", appSecret);
576    verifyParameterPresence("objectType", objectType);
577
578    String[] signedRequestTokens = signedRequest.split("[.]");
579
580    if (signedRequestTokens.length != 2) {
581      throw new FacebookSignedRequestParsingException(format(
582        "Signed request '%s' is expected to be signature and payload strings separated by a '.'", signedRequest));
583    }
584
585    String encodedSignature = signedRequestTokens[0];
586    String urlDecodedSignature = urlDecodeSignedRequestToken(encodedSignature);
587    byte[] signature = decodeBase64(urlDecodedSignature);
588
589    String encodedPayload = signedRequestTokens[1];
590    String urlDecodedPayload = urlDecodeSignedRequestToken(encodedPayload);
591    String payload = StringUtils.toString(decodeBase64(urlDecodedPayload));
592
593    // Convert payload to a JsonObject so we can pull algorithm data out of it
594    JsonObject payloadObject = getJsonMapper().toJavaObject(payload, JsonObject.class);
595
596    if (!payloadObject.contains("algorithm")) {
597      throw new FacebookSignedRequestParsingException("Unable to detect algorithm used for signed request");
598    }
599
600    String algorithm = payloadObject.getString("algorithm", null);
601
602    if (!verifySignedRequest(appSecret, algorithm, encodedPayload, signature)) {
603      throw new FacebookSignedRequestVerificationException(
604        "Signed request verification failed. Are you sure the request was made for the app identified by the app secret you provided?");
605    }
606
607    // Marshal to the user's preferred type.
608    // If the user asked for a JsonObject, send back the one we already parsed.
609    return objectType.equals(JsonObject.class) ? (T) payloadObject : getJsonMapper().toJavaObject(payload, objectType);
610  }

643  /**
644   * Verifies that the signed request is really from Facebook.
645   * 
646   * @param appSecret
647   *          The secret for the app that can verify this signed request.
648   * @param algorithm
649   *          Signature algorithm specified by FB in the decoded payload.
650   * @param encodedPayload
651   *          The encoded payload used to generate a signature for comparison against the provided {@code signature}.
652   * @param signature
653   *          The decoded signature extracted from the signed request. Compared against a signature generated from
654   *          {@code encodedPayload}.
655   * @return {@code true} if the signed request is verified, {@code false} if not.
656   */
657  protected boolean verifySignedRequest(String appSecret, String algorithm, String encodedPayload, byte[] signature) {
658    verifyParameterPresence("appSecret", appSecret);
659    verifyParameterPresence("algorithm", algorithm);
660    verifyParameterPresence("encodedPayload", encodedPayload);
661    verifyParameterPresence("signature", signature);
662
663    // Normalize algorithm name...FB calls it differently than Java does
664    if ("HMAC-SHA256".equalsIgnoreCase(algorithm)) {
665      algorithm = "HMACSHA256";
666    }
667
668    try {
669      Mac mac = Mac.getInstance(algorithm);
670      mac.init(new SecretKeySpec(toBytes(appSecret), algorithm));
671      byte[] payloadSignature = mac.doFinal(toBytes(encodedPayload));
672      return Arrays.equals(signature, payloadSignature);
673    } catch (Exception e) {
674      throw new FacebookSignedRequestVerificationException("Unable to perform signed request verification", e);
675    }
676  }

727  /**
728   * Coordinates the process of executing the API request GET/POST and processing the response we receive from the
729   * endpoint.
730   * 
731   * @param endpoint
732   *          Facebook Graph API endpoint.
733   * @param executeAsPost
734   *          {@code true} to execute the web request as a {@code POST}, {@code false} to execute as a {@code GET}.
735   * @param executeAsDelete
736   *          {@code true} to add a special 'treat this request as a {@code DELETE}' parameter.
737   * @param binaryAttachments
738   *          A list of binary files to include in a {@code POST} request. Pass {@code null} if no attachment should be
739   *          sent.
740   * @param parameters
741   *          Arbitrary number of parameters to send along to Facebook as part of the API call.
742   * @return The JSON returned by Facebook for the API call.
743   * @throws FacebookException
744   *           If an error occurs while making the Facebook API POST or processing the response.
745   */
746  protected String makeRequest(String endpoint, final boolean executeAsPost, final boolean executeAsDelete,
747      final List<BinaryAttachment> binaryAttachments, Parameter... parameters) {
748    verifyParameterLegality(parameters);
749
750    if (executeAsDelete && isHttpDeleteFallback()) {
751      parameters = parametersWithAdditionalParameter(Parameter.with(METHOD_PARAM_NAME, "delete"), parameters);
752    }
753
754    if (!endpoint.startsWith("/")) {
755      endpoint = "/" + endpoint;
756    }
757
758    final String fullEndpoint =
759        createEndpointForApiCall(endpoint, binaryAttachments != null && !binaryAttachments.isEmpty());
760    final String parameterString = toParameterString(parameters);
761
762    return makeRequestAndProcessResponse(new Requestor() {
763      /**
764       * @see com.restfb.DefaultFacebookClient.Requestor#makeRequest()
765       */
766      @Override
767      public Response makeRequest() throws IOException {
768        if (executeAsDelete && !isHttpDeleteFallback()) {
769          return webRequestor.executeDelete(fullEndpoint + "?" + parameterString);
770        }
771
772        if (executeAsPost) {
773          return webRequestor.executePost(fullEndpoint, parameterString, binaryAttachments);
774        }
775
776        return webRequestor.executeGet(fullEndpoint + "?" + parameterString);
777      }
778    });
779  }

812  protected interface Requestor {
813    Response makeRequest() throws IOException;
814  }
815
816  protected String makeRequestAndProcessResponse(Requestor requestor) {
817    Response response;
818
819    // Perform a GET or POST to the API endpoint
820    try {
821      response = requestor.makeRequest();
822    } catch (Exception t) {
823      throw new FacebookNetworkException(t);
824    }
825
826    // If we get any HTTP response code other than a 200 OK or 400 Bad Request
827    // or 401 Not Authorized or 403 Forbidden or 404 Not Found or 500 Internal
828    // Server Error or 302 Not Modified
829    // throw an exception.
830    if (HTTP_OK != response.getStatusCode() && HTTP_BAD_REQUEST != response.getStatusCode()
831        && HTTP_UNAUTHORIZED != response.getStatusCode() && HTTP_NOT_FOUND != response.getStatusCode()
832        && HTTP_INTERNAL_ERROR != response.getStatusCode() && HTTP_FORBIDDEN != response.getStatusCode()
833        && HTTP_NOT_MODIFIED != response.getStatusCode()) {
834      throw new FacebookNetworkException(response.getStatusCode());
835    }
836
837    String json = response.getBody();
838
839    // If the response contained an error code, throw an exception.
840    getFacebookExceptionGenerator().throwFacebookResponseStatusExceptionIfNecessary(json, response.getStatusCode());
841
842    // If there was no response error information and this was a 500 or 401
843    // error, something weird happened on Facebook's end. Bail.
844    if (HTTP_INTERNAL_ERROR == response.getStatusCode() || HTTP_UNAUTHORIZED == response.getStatusCode()) {
845      throw new FacebookNetworkException(response.getStatusCode());
846    }
847
848    return json;
849  }

864  /**
865   * Generate the parameter string to be included in the Facebook API request.
866   * 
867   * @param withJsonParameter
868   *          add additional parameter format with type json
869   * @param parameters
870   *          Arbitrary number of extra parameters to include in the request.
871   * @return The parameter string to include in the Facebook API request.
872   * @throws FacebookJsonMappingException
873   *           If an error occurs when building the parameter string.
874   */
875  protected String toParameterString(boolean withJsonParameter, Parameter... parameters) {
876    if (!isBlank(accessToken)) {
877      parameters = parametersWithAdditionalParameter(Parameter.with(ACCESS_TOKEN_PARAM_NAME, accessToken), parameters);
878    }
879
880    if (!isBlank(accessToken) && !isBlank(appSecret)) {
881      parameters = parametersWithAdditionalParameter(
882        Parameter.with(APP_SECRET_PROOF_PARAM_NAME, obtainAppSecretProof(accessToken, appSecret)), parameters);
883    }
884
885    if (withJsonParameter) {
886      parameters = parametersWithAdditionalParameter(Parameter.with(FORMAT_PARAM_NAME, "json"), parameters);
887    }
888
889    StringBuilder parameterStringBuilder = new StringBuilder();
890    boolean first = true;
891
892    for (Parameter parameter : parameters) {
893      if (first) {
894        first = false;
895      } else {
896        parameterStringBuilder.append("&");
897      }
898
899      parameterStringBuilder.append(urlEncode(parameter.name));
900      parameterStringBuilder.append("=");
901      parameterStringBuilder.append(urlEncodedValueForParameterName(parameter.name, parameter.value));
902    }
903
904    return parameterStringBuilder.toString();
905  }

907  /**
908   * @see com.restfb.BaseFacebookClient#createEndpointForApiCall(java.lang.String,boolean)
909   */
910  @Override
911  protected String createEndpointForApiCall(String apiCall, boolean hasAttachment) {
912    while (apiCall.startsWith("/"))
913      apiCall = apiCall.substring(1);
914
915    String baseUrl = getFacebookGraphEndpointUrl();
916
917    if (hasAttachment && (apiCall.endsWith("/videos") || apiCall.endsWith("/advideos"))) {
918      baseUrl = getFacebookGraphVideoEndpointUrl();
919    } else if (apiCall.endsWith("logout.php")) {
920      baseUrl = getFacebookEndpointUrls().getFacebookEndpoint();
921    }
922
923    return format("%s/%s", baseUrl, apiCall);
924  }
