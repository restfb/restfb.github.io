022package com.restfb;
023
024public enum Version {
025
026  /**
027   * unversiond api
028   */
029  UNVERSIONED(null, false),
030
031  /**
032   * <tt>Graph API 2.5</tt>, available until April 12, 2018
033   *
034   * @since October 7, 2015
035   */
036  VERSION_2_5("v2.5", false),
037
038  /**
039   * <tt>Graph API 2.6</tt>, available until July 13, 2018
040   *
041   * @since April 12, 2016
042   */
043  VERSION_2_6("v2.6", true),
044
045  /**
046   * <tt>Graph API 2.7</tt>, available until October 5, 2018
047   *
048   * @since July 13, 2016
049   */
050  VERSION_2_7("v2.7", true),
051
052  /**
053   * <tt>Graph API 2.8</tt>, available until April 18th, 2019
054   *
055   * @since October 5th, 2016
056   */
057  VERSION_2_8("v2.8", true),
058
059  /**
060   * <tt>Graph API 2.9</tt>, available until July 18th, 2019
061   *
062   * @since April 18th, 2017
063   */
064  VERSION_2_9("v2.9", true),
065
066  /**
067   * <tt>Graph API 2.10</tt>, available until November 7, 2019
068   *
069   * @since July 18th, 2017
070   */
071  VERSION_2_10("v2.10", true),
072
073  /**
074   * <tt>Graph API 2.11</tt>, available until January 30, 2020
075   *
076   * @since November 7, 2017
077   */
078  VERSION_2_11("v2.11", true),
079
080  /**
081   * <tt>Graph API 2.12</tt>, available at least until January, 2020
082   *
083   * @since January 30, 2018
084   */
085  VERSION_2_12("v2.12", true),
086
087  /**
088   * convenience enum to provide simple access to the latest supported Graph API Version.
089   * <p>
090   * the current version is <tt>Graph API 2.12</tt>
091   * </p>
092   */
093  LATEST("v2.12", true);
094
095  private final String urlElement;
096
097  private final boolean newDeviceTokenMethod;
098
099  Version(String urlElement, boolean newDeviceTokenMethod) {
100    this.urlElement = urlElement;
101    this.newDeviceTokenMethod = newDeviceTokenMethod;
102  }
103
104  public String getUrlElement() {
105    return this.urlElement;
106  }
107
108  public boolean isNewDeviceTokenMethod() {
109    return this.newDeviceTokenMethod;
110  }
111
112  public boolean isUrlElementRequired() {
113    return null != this.urlElement;
114  }
115
116  /**
117   * converts a String (for example the url paramter) into a Version object
118   * 
119   * @param urlElement
120   *          String that should
121   * @return the generated version
122   */
123  public static Version getVersionFromString(String urlElement) {
124    if (urlElement != null) {
125      for (Version v : Version.values()) {
126        if (urlElement.equals(v.getUrlElement())) {
127          return v;
128        }
129      }
130    }
131    return UNVERSIONED;
132  }
133}
