041/**
042 * Default implementation of a service that sends HTTP requests to the Facebook API endpoint.
043 * 
044 * @author <a href="http://restfb.com">Mark Allen</a>
045 */
046public class DefaultWebRequestor implements WebRequestor {
047  /**
048   * Arbitrary unique boundary marker for multipart {@code POST}s.
049   */
050  private static final String MULTIPART_BOUNDARY = "**boundarystringwhichwill**neverbeencounteredinthewild**";
051
052  /**
053   * Line separator for multipart {@code POST}s.
054   */
055  private static final String MULTIPART_CARRIAGE_RETURN_AND_NEWLINE = "\r\n";
056
057  /**
058   * Hyphens for multipart {@code POST}s.
059   */
060  private static final String MULTIPART_TWO_HYPHENS = "--";
061
062  /**
063   * Default buffer size for multipart {@code POST}s.
064   */
065  private static final int MULTIPART_DEFAULT_BUFFER_SIZE = 8192;
066
067  /**
068   * By default, how long should we wait for a response (in ms)?
069   */
070  private static final int DEFAULT_READ_TIMEOUT_IN_MS = 180000;
071
072  /**
073   * Logger.
074   */
075  private static final Logger LOGGER = Logger.getLogger("com.restfb.HTTP");
076
077  /**
078   * By default this is true, to prevent breaking existing usage
079   */
080  private boolean autocloseBinaryAttachmentStream = true;
081
082  protected enum HttpMethod {
083    GET, DELETE, POST
084  }

104  /**
105   * @throws java.io.IOException
106   * @see com.restfb.WebRequestor#executePost(java.lang.String, java.lang.String, com.restfb.BinaryAttachment[])
107   */
108  @Override
109  public Response executePost(String url, String parameters, BinaryAttachment... binaryAttachments) throws IOException {
110    if (binaryAttachments == null) {
111      binaryAttachments = new BinaryAttachment[0];
112    }
113
114    if (LOGGER.isLoggable(FINE)) {
115      LOGGER.fine("Executing a POST to " + url + " with parameters "
116          + (binaryAttachments.length > 0 ? "" : "(sent in request body): ") + urlDecode(parameters)
117          + (binaryAttachments.length > 0 ? " and " + binaryAttachments.length + " binary attachment[s]." : ""));
118    }
119
120    HttpURLConnection httpUrlConnection = null;
121    OutputStream outputStream = null;
122    InputStream inputStream = null;
123
124    try {
125      httpUrlConnection = openConnection(new URL(url + (binaryAttachments.length > 0 ? "?" + parameters : "")));
126      httpUrlConnection.setReadTimeout(DEFAULT_READ_TIMEOUT_IN_MS);
127
128      // Allow subclasses to customize the connection if they'd like to - set
129      // their own headers, timeouts, etc.
130      customizeConnection(httpUrlConnection);
131
132      httpUrlConnection.setRequestMethod("POST");
133      httpUrlConnection.setDoOutput(true);
134      httpUrlConnection.setUseCaches(false);
135
136      if (binaryAttachments.length > 0) {
137        httpUrlConnection.setRequestProperty("Connection", "Keep-Alive");
138        httpUrlConnection.setRequestProperty("Content-Type", "multipart/form-data;boundary=" + MULTIPART_BOUNDARY);
139      }
140
141      httpUrlConnection.connect();
142      outputStream = httpUrlConnection.getOutputStream();
143
144      // If we have binary attachments, the body is just the attachments and the
145      // other parameters are passed in via the URL.
146      // Otherwise the body is the URL parameter string.
147      if (binaryAttachments.length > 0) {
148        for (BinaryAttachment binaryAttachment : binaryAttachments) {
149          StringBuilder stringBuilder = new StringBuilder();
150
151          stringBuilder.append(MULTIPART_TWO_HYPHENS).append(MULTIPART_BOUNDARY)
152            .append(MULTIPART_CARRIAGE_RETURN_AND_NEWLINE).append("Content-Disposition: form-data; name=\"")
153            .append(createFormFieldName(binaryAttachment)).append("\"; filename=\"")
154            .append(binaryAttachment.getFilename()).append("\"");
155
156          if (binaryAttachment.getContentType() != null && binaryAttachment.getContentType().length() > 0) {
157            stringBuilder.append(MULTIPART_CARRIAGE_RETURN_AND_NEWLINE).append("Content-Type: ")
158              .append(binaryAttachment.getContentType());
159          }
160
161          stringBuilder.append(MULTIPART_CARRIAGE_RETURN_AND_NEWLINE).append(MULTIPART_CARRIAGE_RETURN_AND_NEWLINE);
162
163          outputStream.write(stringBuilder.toString().getBytes(ENCODING_CHARSET));
164
165          write(binaryAttachment.getData(), outputStream, MULTIPART_DEFAULT_BUFFER_SIZE);
166
167          outputStream.write((MULTIPART_CARRIAGE_RETURN_AND_NEWLINE + MULTIPART_TWO_HYPHENS + MULTIPART_BOUNDARY
168              + MULTIPART_TWO_HYPHENS + MULTIPART_CARRIAGE_RETURN_AND_NEWLINE).getBytes(ENCODING_CHARSET));
169        }
170      } else {
171        outputStream.write(parameters.getBytes(ENCODING_CHARSET));
172      }
173
174      if (LOGGER.isLoggable(FINER)) {
175        LOGGER.log(FINER, "Response headers: {0}", httpUrlConnection.getHeaderFields());
176      }
177
178      try {
179        inputStream =
180            httpUrlConnection.getResponseCode() != HTTP_OK ? httpUrlConnection.getErrorStream() : httpUrlConnection
181              .getInputStream();
182      } catch (IOException e) {
183        if (LOGGER.isLoggable(WARNING)) {
184          LOGGER.log(WARNING, "An error occurred while POSTing to {0}: {1}", new Object[] { url, e });
185        }
186      }
187
188      return new Response(httpUrlConnection.getResponseCode(), fromInputStream(inputStream));
189    } finally {
190      if (autocloseBinaryAttachmentStream && binaryAttachments.length > 0) {
191        for (BinaryAttachment binaryAttachment : binaryAttachments) {
192          closeQuietly(binaryAttachment.getData());
193        }
194      }
195
196      closeQuietly(outputStream);
197      closeQuietly(httpUrlConnection);
198    }
199  }

201  /**
202   * Given a {@code url}, opens and returns a connection to it.
203   * <p>
204   * If you'd like to pipe your connection through a proxy, this is the place to do so.
205   * 
206   * @param url
207   *          The URL to connect to.
208   * @return A connection to the URL.
209   * @throws IOException
210   *           If an error occurs while establishing the connection.
211   * @since 1.6.3
212   */
213  protected HttpURLConnection openConnection(URL url) throws IOException {
214    return (HttpURLConnection) url.openConnection();
215  }

224   * @param connection
225   *          The connection to customize.
226   */

229  /**
230   * Attempts to cleanly close a resource, swallowing any exceptions that might occur since there's no way to recover
231   * anyway.
232   * <p>
233   * It's OK to pass {@code null} in, this method will no-op in that case.
234   * 
235   * @param closeable
236   *          The resource to close.
237   */
238  protected void closeQuietly(Closeable closeable) {
239    if (closeable == null) {
240      return;
241    }
242    try {
243      closeable.close();
244    } catch (Throwable t) {
245      if (LOGGER.isLoggable(WARNING)) {
246        LOGGER.log(WARNING, "Unable to close {0}: {1}", new Object[] { closeable, t });
247      }
248    }
249  }
250
251  /**
252   * Attempts to cleanly close an {@code HttpURLConnection}, swallowing any exceptions that might occur since there's no
253   * way to recover anyway.
254   * <p>
255   * It's OK to pass {@code null} in, this method will no-op in that case.
256   * 
257   * @param httpUrlConnection
258   *          The connection to close.
259   */
260  protected void closeQuietly(HttpURLConnection httpUrlConnection) {
261    if (httpUrlConnection == null) {
262      return;
263    }
264    try {
265      httpUrlConnection.disconnect();
266    } catch (Throwable t) {
267      if (LOGGER.isLoggable(WARNING)) {
268        LOGGER.log(WARNING, "Unable to disconnect {0}: {1}", new Object[] { httpUrlConnection, t });
269      }
270    }
271  }

273  /**
274   * Writes the contents of the {@code source} stream to the {@code destination} stream using the given
275   * {@code bufferSize}.
276   * 
277   * @param source
278   *          The source stream to copy from.
279   * @param destination
280   *          The destination stream to copy to.
281   * @param bufferSize
282   *          The size of the buffer to use during the copy operation.
283   * @throws IOException
284   *           If an error occurs when reading from {@code source} or writing to {@code destination}.
285   * @throws NullPointerException
286   *           If either {@code source} or @{code destination} is {@code null}.
287   */
288  protected void write(InputStream source, OutputStream destination, int bufferSize) throws IOException {
289    if (source == null || destination == null) {
290      throw new NullPointerException("Must provide non-null source and destination streams.");
291    }
292
293    int read;
294    byte[] chunk = new byte[bufferSize];
295    while ((read = source.read(chunk)) > 0)
296      destination.write(chunk, 0, read);
297  }

299  /**
300   * Creates the form field name for the binary attachment filename by stripping off the file extension - for example,
301   * the filename "test.png" would return "test".
302   * 
303   * @param binaryAttachment
304   *          The binary attachment for which to create the form field name.
305   * @return The form field name for the given binary attachment.
306   */
307  protected String createFormFieldName(BinaryAttachment binaryAttachment) {
308    String name = binaryAttachment.getFilename();
309    int fileExtensionIndex = name.lastIndexOf(".");
310    return fileExtensionIndex > 0 ? name.substring(0, fileExtensionIndex) : name;
311  }

333  @Override
334  public Response executeDelete(String url) throws IOException {
335    return execute(url, HttpMethod.DELETE);

338  private Response execute(String url, HttpMethod httpMethod) throws IOException {
339    if (LOGGER.isLoggable(FINE)) {
340      LOGGER.log(FINE, "Making a {0} request to {1}", new Object[] { httpMethod.name(), url });
341    }
342
343    HttpURLConnection httpUrlConnection = null;
344    InputStream inputStream = null;
345
346    try {
347      httpUrlConnection = openConnection(new URL(url));
348      httpUrlConnection.setReadTimeout(DEFAULT_READ_TIMEOUT_IN_MS);
349      httpUrlConnection.setUseCaches(false);
350      httpUrlConnection.setRequestMethod(httpMethod.name());
351
352      // Allow subclasses to customize the connection if they'd like to - set
353      // their own headers, timeouts, etc.
354      customizeConnection(httpUrlConnection);
355
356      httpUrlConnection.connect();
357
358      if (LOGGER.isLoggable(FINER)) {
359        LOGGER.log(FINER, "Response headers: {0}", httpUrlConnection.getHeaderFields());
360      }
361
362      Response response = fetchResponse(inputStream, httpUrlConnection);
363
364      if (LOGGER.isLoggable(FINE)) {
365        LOGGER.log(FINE, "Facebook responded with {0}", response);
366      }
367
368      return response;
369    } finally {
370      closeQuietly(httpUrlConnection);
371    }
372  }
373
374  protected Response fetchResponse(InputStream inputStream, HttpURLConnection httpUrlConnection) throws IOException {
375    try {
376      inputStream =
377          httpUrlConnection.getResponseCode() != HTTP_OK ? httpUrlConnection.getErrorStream() : httpUrlConnection
378            .getInputStream();
379    } catch (IOException e) {
380      if (LOGGER.isLoggable(WARNING)) {
381        LOGGER.warning(format("An error occurred while making a " + httpUrlConnection.getRequestMethod()
382            + " request to %s: %s", httpUrlConnection.getURL(), e));
383      }
384    }
385
386    return new Response(httpUrlConnection.getResponseCode(), fromInputStream(inputStream));
387  }
