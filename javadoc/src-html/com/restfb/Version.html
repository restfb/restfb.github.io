023package com.restfb;
024
025public enum Version {
026  /**
027   * unversiond api
028   */
029  UNVERSIONED(null),
030  /**
031   * graph api 1.0, available until April 30, 2015
032   * 
033   * @since April 21, 2010
034   * @deprecated use VERSION_2_0 instead
035   */
036  @Deprecated
037  VERSION_1_0("v1.0"),
038  /**
039   * graph api 2.0, available until August 7, 2016
040   * 
041   * @since April 30, 2014
042   */
043  VERSION_2_0("v2.0"),
044  /**
045   * graph api 2.1, available until October 30, 2016
046   * 
047   * @since August 7, 2014
048   */
049  VERSION_2_1("v2.1"),
050  /**
051   * graph api 2.2, available until March 25, 2017
052   * 
053   * @since October 30, 2014
054   */
055  VERSION_2_2("v2.2"),
056  /**
057   * graph api 2.3, available at least until March 25, 2017
058   * 
059   * @since March 25, 2015
060   */
061  VERSION_2_3("v2.3");
062
063  private final String urlElement;
064
065  Version(String urlElement) {
066    this.urlElement = urlElement;
067  }
068
069  public String getUrlElement() {
070    return this.urlElement;
071  }
072
073  public boolean isUrlElementRequired() {
074    return null != this.urlElement;
075  }
076}
