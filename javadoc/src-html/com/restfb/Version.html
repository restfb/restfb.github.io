022package com.restfb;
023
024public enum Version {
025
026  /**
027   * unversiond api
028   */
029  UNVERSIONED(null, false),
030
031  /**
032   * <tt>Graph API 1.0</tt>, available until April 30, 2015
033   * 
034   * @since April 21, 2010
035   * @deprecated use {@link Version#VERSION_2_2} instead
036   */
037  @Deprecated
038  VERSION_1_0("v1.0", false),
039
040  /**
041   * <tt>Graph API 2.0</tt>, available until August 7, 2016
042   *
043   * @since April 30, 2014
044   * @deprecated use {@link Version#VERSION_2_2} instead
045   */
046  @Deprecated
047  VERSION_2_0("v2.0", false),
048
049  /**
050   * <tt>Graph API 2.1</tt>, available until October 30, 2016
051   *
052   * @since August 7, 2014
053   * @deprecated use {@link Version#VERSION_2_2} instead
054   */
055  @Deprecated
056  VERSION_2_1("v2.1", false),
057
058  /**
059   * <tt>Graph API 2.2</tt>, available until March 25, 2017
060   *
061   * @since October 30, 2014
062   */
063  VERSION_2_2("v2.2", false),
064
065  /**
066   * <tt>Graph API 2.3</tt>, available until July 8, 2017
067   *
068   * @since March 25, 2015
069   */
070  VERSION_2_3("v2.3", false),
071
072  /**
073   * <tt>Graph API 2.4</tt>, available until October 7, 2017
074   *
075   * @since July 8, 2015
076   */
077  VERSION_2_4("v2.4", false),
078
079  /**
080   * <tt>Graph API 2.5</tt>, available until April 12, 2018
081   *
082   * @since October 7, 2015
083   */
084  VERSION_2_5("v2.5", false),
085
086  /**
087   * <tt>Graph API 2.6</tt>, available until July 13, 2018
088   *
089   * @since April 12, 2016
090   */
091  VERSION_2_6("v2.6", true),
092
093  /**
094   * <tt>Graph API 2.7</tt>, available until October 5, 2018
095   *
096   * @since July 13, 2016
097   */
098  VERSION_2_7("v2.7", true),
099
100  /**
101   * <tt>Graph API 2.8</tt>, available at least until October 2018
102   *
103   * @since October 5th, 2016
104   */
105  VERSION_2_8("v2.8", true),
106
107  /**
108   * convenience enum to provide simple access to the latest supported Graph API Version.
109   * <p>
110   * the current version is <tt>Graph API 2.8</tt>
111   * </p>
112   */
113  LATEST("v2.8", true);
114
115  private final String urlElement;
116
117  private final boolean newDeviceTokenMethod;
118
119  Version(String urlElement, boolean newDeviceTokenMethod) {
120    this.urlElement = urlElement;
121    this.newDeviceTokenMethod = newDeviceTokenMethod;
122  }
123
124  public String getUrlElement() {
125    return this.urlElement;
126  }
127
128  public boolean isNewDeviceTokenMethod() {
129    return this.newDeviceTokenMethod;
130  }
131
132  public boolean isUrlElementRequired() {
133    return null != this.urlElement;
134  }
135
136  /**
137   * converts a String (for example the url paramter) into a Version object
138   * 
139   * @param urlElement
140   *          String that should
141   * @return the generated version
142   */
143  public static Version getVersionFromString(String urlElement) {
144    if (urlElement != null) {
145      for (Version v : Version.values()) {
146        if (urlElement.equals(v.getUrlElement())) {
147          return v;
148        }
149      }
150    }
151    return UNVERSIONED;
152  }
153}
