236  @Override
237  public boolean deleteObject(String object, Parameter... parameters) {
238    verifyParameterPresence("object", object);
239
240    String responseString = makeRequest(object, true, true, null, parameters);
241
242    try {
243      JsonValue jObj = Json.parse(responseString);
244      boolean success = false;
245      if (jObj.isObject()) {
246        if (jObj.asObject().contains("success")) {
247          success = jObj.asObject().get("success").asBoolean();
248        }
249        if (jObj.asObject().contains("result")) {
250          success = jObj.asObject().get("result").asString().contains("Successfully deleted");
251        }
252      } else {
253        success = jObj.asBoolean();
254      }
255      return success;
256    } catch (ParseException jex) {
257      CLIENT_LOGGER.trace("no valid JSON returned while deleting a object, using returned String instead", jex);
258      return "true".equals(responseString);
259    }
260  }

307   */
308  @Override
309  @SuppressWarnings("unchecked")
310  public <T> T fetchObjects(List<String> ids, Class<T> objectType, Parameter... parameters) {
311    verifyParameterPresence("ids", ids);
312    verifyParameterPresence(CONNECTION_TYPE, objectType);
313    requireNotEmpty(ids, "The list of IDs cannot be empty.");
314
315    if (Stream.of(parameters).anyMatch(p -> IDS_PARAM_NAME.equals(p.name))) {
316      throw new IllegalArgumentException("You cannot specify the '" + IDS_PARAM_NAME + "' URL parameter yourself - "
317          + "RestFB will populate this for you with the list of IDs you passed to this method.");
318    }
319
320    JsonArray idArray = new JsonArray();
321
322    // Normalize the IDs
323    for (String id : ids) {
324      throwIAEonBlankId(id);
325      idArray.add(id.trim());
326    }
327
328    try {
329      String jsonString = makeRequest("",
330        parametersWithAdditionalParameter(Parameter.with(IDS_PARAM_NAME, idArray.toString()), parameters));
331
332      return jsonMapper.toJavaObject(jsonString, objectType);
333    } catch (ParseException e) {
334      throw new FacebookJsonMappingException("Unable to map connection JSON to Java objects", e);
335    }
336  }

340      throw new IllegalArgumentException("The list of IDs cannot contain blank strings.");

558  @Override
559  @SuppressWarnings("unchecked")
560  public <T> T parseSignedRequest(String signedRequest, String appSecret, Class<T> objectType) {
561    verifyParameterPresence("signedRequest", signedRequest);
562    verifyParameterPresence(APP_SECRET, appSecret);
563    verifyParameterPresence("objectType", objectType);
564
565    String[] signedRequestTokens = signedRequest.split("[.]");
566
567    if (signedRequestTokens.length != 2) {
568      throw new FacebookSignedRequestParsingException(format(
569        "Signed request '%s' is expected to be signature and payload strings separated by a '.'", signedRequest));
570    }
571
572    String encodedSignature = signedRequestTokens[0];
573    String urlDecodedSignature = urlDecodeSignedRequestToken(encodedSignature);
574    byte[] signature = decodeBase64(urlDecodedSignature);
575
576    String encodedPayload = signedRequestTokens[1];
577    String urlDecodedPayload = urlDecodeSignedRequestToken(encodedPayload);
578    String payload = StringUtils.toString(decodeBase64(urlDecodedPayload));
579
580    // Convert payload to a JsonObject so we can pull algorithm data out of it
581    JsonObject payloadObject = getJsonMapper().toJavaObject(payload, JsonObject.class);
582
583    if (!payloadObject.contains(ALGORITHM)) {
584      throw new FacebookSignedRequestParsingException("Unable to detect algorithm used for signed request");
585    }
586
587    String algorithm = payloadObject.getString(ALGORITHM, null);
588
589    if (!verifySignedRequest(appSecret, algorithm, encodedPayload, signature)) {
590      throw new FacebookSignedRequestVerificationException(
591        "Signed request verification failed. Are you sure the request was made for the app identified by the app secret you provided?");
592    }
593
594    // Marshal to the user's preferred type.
595    // If the user asked for a JsonObject, send back the one we already parsed.
596    return objectType.equals(JsonObject.class) ? (T) payloadObject : getJsonMapper().toJavaObject(payload, objectType);
597  }

630  /**
631   * Verifies that the signed request is really from Facebook.
632   * 
633   * @param appSecret
634   *          The secret for the app that can verify this signed request.
635   * @param algorithm
636   *          Signature algorithm specified by FB in the decoded payload.
637   * @param encodedPayload
638   *          The encoded payload used to generate a signature for comparison against the provided {@code signature}.
639   * @param signature
640   *          The decoded signature extracted from the signed request. Compared against a signature generated from
641   *          {@code encodedPayload}.
642   * @return {@code true} if the signed request is verified, {@code false} if not.
643   */
644  protected boolean verifySignedRequest(String appSecret, String algorithm, String encodedPayload, byte[] signature) {
645    verifyParameterPresence(APP_SECRET, appSecret);
646    verifyParameterPresence(ALGORITHM, algorithm);
647    verifyParameterPresence("encodedPayload", encodedPayload);
648    verifyParameterPresence("signature", signature);
649
650    // Normalize algorithm name...FB calls it differently than Java does
651    if ("HMAC-SHA256".equalsIgnoreCase(algorithm)) {
652      algorithm = "HMACSHA256";
653    }
654
655    try {
656      Mac mac = Mac.getInstance(algorithm);
657      mac.init(new SecretKeySpec(toBytes(appSecret), algorithm));
658      byte[] payloadSignature = mac.doFinal(toBytes(encodedPayload));
659      return Arrays.equals(signature, payloadSignature);
660    } catch (Exception e) {
661      throw new FacebookSignedRequestVerificationException("Unable to perform signed request verification", e);
662    }
663  }

726  /**
727   * Coordinates the process of executing the API request GET/POST and processing the response we receive from the
728   * endpoint.
729   * 
730   * @param endpoint
731   *          Facebook Graph API endpoint.
732   * @param executeAsPost
733   *          {@code true} to execute the web request as a {@code POST}, {@code false} to execute as a {@code GET}.
734   * @param executeAsDelete
735   *          {@code true} to add a special 'treat this request as a {@code DELETE}' parameter.
736   * @param binaryAttachments
737   *          A list of binary files to include in a {@code POST} request. Pass {@code null} if no attachment should be
738   *          sent.
739   * @param parameters
740   *          Arbitrary number of parameters to send along to Facebook as part of the API call.
741   * @return The JSON returned by Facebook for the API call.
742   * @throws FacebookException
743   *           If an error occurs while making the Facebook API POST or processing the response.
744   */
745  protected String makeRequest(String endpoint, final boolean executeAsPost, final boolean executeAsDelete,
746      final List<BinaryAttachment> binaryAttachments, Body body, Parameter... parameters) {
747    verifyParameterLegality(parameters);
748
749    if (executeAsDelete && isHttpDeleteFallback()) {
750      parameters = parametersWithAdditionalParameter(Parameter.with(METHOD_PARAM_NAME, "delete"), parameters);
751    }
752
753    if (!endpoint.startsWith("/")) {
754      endpoint = "/" + endpoint;
755    }
756
757    boolean hasAttachment = binaryAttachments != null && !binaryAttachments.isEmpty();
758    boolean hasReel = hasAttachment && binaryAttachments.get(0).isFacebookReel();
759
760    final String fullEndpoint =
761        createEndpointForApiCall(endpoint, hasAttachment, hasReel);
762    final String parameterString = toParameterString(parameters);
763
764    String headerAccessToken = (hasReel) ? accessToken : getHeaderAccessToken();
765
766    return makeRequestAndProcessResponse(() -> {
767      WebRequestor.Request request = new WebRequestor.Request(fullEndpoint, headerAccessToken, parameterString);
768      if (executeAsDelete && !isHttpDeleteFallback()) {
769        return webRequestor.executeDelete(request);
770      }
771
772      if (executeAsPost) {
773        request.setBinaryAttachments(binaryAttachments);
774        request.setBody(body);
775        return webRequestor.executePost(request);
776      }
777
778      return webRequestor.executeGet(request);
779    });
780  }

821  protected interface Requestor {
822    Response makeRequest() throws IOException;
823  }
824
825  protected String makeRequestAndProcessResponse(Requestor requestor) {
826    Response response;
827
828    // Perform a GET or POST to the API endpoint
829    try {
830      response = requestor.makeRequest();
831    } catch (Exception t) {
832      throw new FacebookNetworkException(t);
833    }
834
835    // If we get any HTTP response code other than a 200 OK or 400 Bad Request
836    // or 401 Not Authorized or 403 Forbidden or 404 Not Found or 500 Internal
837    // Server Error or 302 Not Modified
838    // throw an exception.
839    if (HTTP_OK != response.getStatusCode() && HTTP_BAD_REQUEST != response.getStatusCode()
840        && HTTP_UNAUTHORIZED != response.getStatusCode() && HTTP_NOT_FOUND != response.getStatusCode()
841        && HTTP_INTERNAL_ERROR != response.getStatusCode() && HTTP_FORBIDDEN != response.getStatusCode()
842        && HTTP_NOT_MODIFIED != response.getStatusCode()) {
843      throw new FacebookNetworkException(response.getStatusCode());
844    }
845
846    String json = response.getBody();
847
848    try {
849      // If the response contained an error code, throw an exception.
850      getFacebookExceptionGenerator().throwFacebookResponseStatusExceptionIfNecessary(json, response.getStatusCode());
851    } catch (FacebookErrorMessageException feme) {
852      Optional.ofNullable(getWebRequestor()).map(WebRequestor::getDebugHeaderInfo).ifPresent(feme::setDebugHeaderInfo);
853      throw feme;
854    }
855
856    // If there was no response error information and this was a 500 or 401
857    // error, something weird happened on Facebook's end. Bail.
858    if (HTTP_INTERNAL_ERROR == response.getStatusCode() || HTTP_UNAUTHORIZED == response.getStatusCode()) {
859      throw new FacebookNetworkException(response.getStatusCode());
860    }
861
862    return json;
863  }

878  /**
879   * Generate the parameter string to be included in the Facebook API request.
880   * 
881   * @param withJsonParameter
882   *          add additional parameter format with type json
883   * @param parameters
884   *          Arbitrary number of extra parameters to include in the request.
885   * @return The parameter string to include in the Facebook API request.
886   * @throws FacebookJsonMappingException
887   *           If an error occurs when building the parameter string.
888   */
889  protected String toParameterString(boolean withJsonParameter, Parameter... parameters) {
890    if (!isBlank(accessToken) && !accessTokenInHeader) {
891      parameters = parametersWithAdditionalParameter(Parameter.with(ACCESS_TOKEN_PARAM_NAME, accessToken), parameters);
892    }
893
894    if (!isBlank(accessToken) && !isBlank(appSecret)) {
895      parameters = parametersWithAdditionalParameter(
896        Parameter.with(APP_SECRET_PROOF_PARAM_NAME, obtainAppSecretProof(accessToken, appSecret)), parameters);
897    }
898
899    if (withJsonParameter) {
900      parameters = parametersWithAdditionalParameter(Parameter.with(FORMAT_PARAM_NAME, "json"), parameters);
901    }
902
903    return Stream.of(parameters).map(p -> urlEncode(p.name) + "=" + urlEncodedValueForParameterName(p.name, p.value))
904      .collect(Collectors.joining("&"));
905  }

907  /**
908   * @see com.restfb.BaseFacebookClient#createEndpointForApiCall(java.lang.String,boolean,boolean)
909   */
910  @Override
911  protected String createEndpointForApiCall(String apiCall, boolean hasAttachment, boolean hasReel) {
912    while (apiCall.startsWith("/")) {
913      apiCall = apiCall.substring(1);
914    }
915
916    String baseUrl = getFacebookGraphEndpointUrl();
917
918    if (hasAttachment && hasReel) {
919      baseUrl = getFacebookReelsUploadEndpointUrl();
920    } else if (hasAttachment && (apiCall.endsWith("/videos") || apiCall.endsWith("/advideos"))) {
921      baseUrl = getFacebookGraphVideoEndpointUrl();
922    } else if (apiCall.endsWith("logout.php")) {
923      baseUrl = getFacebookEndpointUrls().getFacebookEndpoint();
924    }
925
926    return format("%s/%s", baseUrl, apiCall);
927  }
