022package com.restfb.types.webhook.messaging;
023
024import java.util.*;
025
026/**
027 * Represents the app roles as used here:
028 * <a href="https://developers.facebook.com/docs/messenger-platform/reference/webhook-events/app_role">App Roles on
029 * Facebook</a>
030 */
031public class AppRoles implements InnerMessagingItem {
032
033  private Map<String, List<String>> roles = new HashMap<>();
034
035  /**
036   * Get a set of app ids
037   * 
038   * @return set of app ids
039   */
040  public Set<String> getAppIds() {
041    return Collections.unmodifiableSet(roles.keySet());
042  }
043
044  /**
045   * get the roles to the given app id
046   * 
047   * @param appId
048   * @return List of app roles
049   */
050  public List<String> getRoles(String appId) {
051    if (roles.containsKey(appId)) {
052      return Collections.unmodifiableList(roles.get(appId));
053    } else {
054      return null;
055    }
056  }
057
058  void addRoles(String appId, List<String> roles) {
059    this.roles.put(appId, roles);
060  }
061}
