025import java.util.ArrayList;
026import java.util.List;
027import com.restfb.Facebook;
028import com.restfb.types.AbstractFacebookType;
029import com.restfb.types.send.AbstractButton;

033/**
034 * Represents the media template element that is used with the attachment id as defined
035 * <a href="https://developers.facebook.com/docs/messenger-platform/send-messages/template/media#attachment">here</a>
036 */
037public class MediaTemplateAttachmentElement extends AbstractFacebookType implements MediaAttachment.MediaTemplateElement {
038  @Facebook("media_type")
039  private String mediaType;
040  @Facebook("attachment_id")
041  private String attachmentId;
042  @Facebook
043  private List<AbstractButton> buttons;
044
045  public MediaTemplateAttachmentElement(MediaAttachment.MediaType mediaType, String attachmentId) {
046    this.mediaType = mediaType.name().toLowerCase();
047    this.attachmentId = attachmentId;
048  }
049
050  @Override
051  public void addButton(AbstractButton button) {
052    if (buttons == null) {
053      buttons = new ArrayList<>();
054    }
055    buttons.add(button);
056  }
057
058  @java.lang.SuppressWarnings("all")
059  public String getMediaType() {
060    return this.mediaType;
061  }
062
063  @java.lang.SuppressWarnings("all")
064  public String getAttachmentId() {
065    return this.attachmentId;
066  }
067
068  @java.lang.SuppressWarnings("all")
069  public List<AbstractButton> getButtons() {
070    return this.buttons;
071  }
072}
