022package com.restfb.exception;
023
024import com.restfb.DebugHeaderInfo;
025import com.restfb.json.JsonObject;
026
027/**
028 * Abstract class to provide access to the JSON Facebook provides in case of an error
029 *
030 * Sometime a developer needs to access the plain error to get a more in depth view to the error.
031 */
032public abstract class FacebookErrorMessageException extends FacebookException {
033
034  private JsonObject rawErrorJson;
035
036  private DebugHeaderInfo debugHeaderInfo;
037
038  protected FacebookErrorMessageException(String message) {
039    super(message);
040  }
041
042  protected FacebookErrorMessageException(String message, Throwable cause) {
043    super(message, cause);
044  }
045
046  /**
047   * return the raw error as JSON, may be <code>null</code>
048   * 
049   * @return raw error
050   */
051  public JsonObject getRawErrorJson() {
052    return rawErrorJson;
053  }
054
055  protected void setRawErrorJson(JsonObject rawError) {
056    rawErrorJson = rawError;
057  }
058
059  /**
060   * returns the debug header info that is connected with this Facebook call.
061   *
062   * you can get information like trace ids, limits and more.
063   * 
064   * @return the DebugHeaderInfo or null
065   */
066  public DebugHeaderInfo getDebugHeaderInfo() {
067    return debugHeaderInfo;
068  }
069
070  public void setDebugHeaderInfo(DebugHeaderInfo debugHeaderInfo) {
071    this.debugHeaderInfo = debugHeaderInfo;
072  }
073}
