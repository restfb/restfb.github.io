023package com.restfb.scope;
024
025public enum UserDataPermissions implements FacebookPermissions {
026
027  /**
028   * Provides access to a person's personal description (the 'About Me' section on their Profile) through the bio
029   * property on the User object.
030   * 
031   * This permission does not give access to a person's public profile data. A person's name, profile picture, locale,
032   * age range and gender are included by default with the public_profile permission.
033   * 
034   * <strong>Review</strong>
035   * 
036   * If your app requests this permission Facebook will have to review how your app uses it.
037   */
038  USER_ABOUT_ME("user_about_me"), //
039
040  /**
041   * Provides access to all common books actions published by any app the person has used. This includes books they've
042   * read, want to read, rated or quoted.
043   * 
044   * <strong>Review</strong>
045   * 
046   * If your app requests this permission Facebook will have to review how your app uses it.
047   */
048  USER_ACTIONS_BOOKS("user_actions.books"), //
049
050  /**
051   * Provides access to all common Open Graph fitness actions published by any app the person has used. This includes
052   * runs, walks and bikes actions.
053   * 
054   * <strong>Review</strong>
055   * 
056   * If your app requests this permission Facebook will have to review how your app uses it.
057   */
058  USER_ACTIONS_FITNESS("user_actions.fitness"), //
059
060  /**
061   * Provides access to all common Open Graph music actions published by any app the person has used. This includes
062   * songs they've listened to, and playlists they've created.
063   * 
064   * <strong>Review</strong>
065   * 
066   * If your app requests this permission Facebook will have to review how your app uses it.
067   */
068  USER_ACTIONS_MUSIC("user_actions.music"), //
069
070  /**
071   * Provides access to all common Open Graph news actions published by any app the person has used which publishes
072   * these actions. This includes news articles they've read or news articles they've published.
073   * 
074   * <strong>Review</strong>
075   * 
076   * If your app requests this permission Facebook will have to review how your app uses it.
077   */
078  USER_ACTIONS_NEWS("user_actions.news"), //
079
080  /**
081   * Provides access to all common Open Graph video actions published by any app the person has used which publishes
082   * these actions. This includes videos they've watched, videos they've rated and videos they want to watch.
083   * 
084   * <strong>Review</strong>
085   * 
086   * If your app requests this permission Facebook will have to review how your app uses it.
087   */
088  USER_ACTIONS_VIDEO("user_actions.video"), //
089
090  /**
091   * Provides access to a person's list of activities as listed on their Profile. This is a subset of the pages they
092   * have liked, where those pages represent particular interests. This information is accessed through the activities
093   * connection on the user node.
094   * 
095   * <strong>Review</strong>
096   * 
097   * If your app requests this permission Facebook will have to review how your app uses it.
098   */
099  USER_ACTIVITIES("user_activities"), //
100
101  /**
102   * Access the date and month of a person's birthday. This may or may not include the person's year of birth, dependent
103   * upon their privacy settings and the access token being used to query this field.
104   * 
105   * Please note most integrations will only need age_range which comes as part of the public_profile permission.
106   * 
107   * <strong>Review</strong>
108   * 
109   * If your app requests this permission Facebook will have to review how your app uses it.
110   */
111  USER_BIRTHDAY("user_birthday"), //
112
113  /**
114   * Provides access to a person's education history through the education field on the User object.
115   * 
116   * <strong>Review</strong>
117   * 
118   * If your app requests this permission Facebook will have to review how your app uses it.
119   */
120  USER_EDUCATION_HISTORY("user_education_history"), //
121
122  /**
123   * Provides read-only access to the Events a person is hosting or has RSVP'd to.
124   * 
125   * <strong>Review</strong>
126   * 
127   * If your app requests this permission Facebook will have to review how your app uses it.
128   */
129  USER_EVENTS("user_events"), //
130
131  /**
132   * Provides access the list of friends that also use your app. These friends can be found on the friends edge on the
133   * user object.
134   * 
135   * In order for a person to show up in one person's friend list, both people must have decided to share their list of
136   * friends with your app and not disabled that permission during login. Also both friends must have been asked for
137   * <code>user_friends</code> during the login process.
138   * 
139   * <strong>Review</strong>
140   * 
141   * Your app may use this permission without review from Facebook.
142   */
143  USER_FRIENDS("user_friends"), //
144
145  /**
146   * Provides access to read a person's game activity (scores, achievements) in any game the person has played.
147   * 
148   * <strong>Review</strong>
149   * 
150   * If your app requests this permission Facebook will have to review how your app uses it.
151   */
152  USER_GAMES_ACTIVITY("user_games_activity"), //
153
154  /**
155   * Enables your app to read the Groups a person is a member of through the groups edge on the User object.
156   * 
157   * This permission does not allow you to create groups on behalf of a person. It is not possible to create groups via
158   * the Graph API.
159   * 
160   * <strong>Review</strong>
161   * 
162   * If your app requests this permission Facebook will have to review how your app uses it.
163   */
164  USER_GROUPS("user_groups"), //
165
166  /**
167   * Provides access to a person's hometown location through the hometown field on the User object. This is set by the
168   * user on the Profile.
169   * 
170   * <strong>Review</strong>
171   * 
172   * If your app requests this permission Facebook will have to review how your app uses it.
173   */
174  USER_HOMETOWN("user_hometown"), //
175
176  /**
177   * Provides access to the list of interests in a person's Profile. This is a subset of the pages they have liked which
178   * represent particular interests.
179   * 
180   * <strong>Review</strong>
181   * 
182   * If your app requests this permission Facebook will have to review how your app uses it.
183   */
184  USER_INTERESTS("user_interests"), //
185
186  /**
187   * Provides access to the list of all Facebook Pages and Open Graph objects that a person has liked. This list is
188   * available through the likes edge on the User object.
189   * 
190   * <strong>Review</strong>
191   * 
192   * If your app requests this permission Facebook will have to review how your app uses it.
193   */
194  USER_LIKES("user_likes"), //
195
196  /**
197   * Provides access to a person's current city through the location field on the User object. The current city is set
198   * by a person on their Profile.
199   * 
200   * The current city is not necessarily the same as a person's hometown.
201   * 
202   * <strong>Review</strong>
203   * 
204   * If your app requests this permission Facebook will have to review how your app uses it.
205   */
206  USER_LOCATION("user_location"), //
207
208  /**
209   * Provides access to the photos a person has uploaded or been tagged in. This is available through the photos edge on
210   * the User object.
211   * 
212   * <strong>Review</strong>
213   * 
214   * If your app requests this permission Facebook will have to review how your app uses it.
215   */
216  USER_PHOTOS("user_photos"), //
217
218  /**
219   * Provides access to the posts on a person's Timeline.
220   * 
221   * Includes their own posts, posts they are tagged in, and posts other people make on their Timeline.
222   * 
223   * <strong>Review</strong>
224   * 
225   * If your app requests this permission Facebook will have to review how your app uses it.
226   * 
227   * @since 1.10.0
228   */
229  USER_POSTS("user_posts"), //
230
231  /**
232   * Provides access to a person's relationship interests as the <code>interested_in</code> field on the User object.
233   * 
234   * <strong>Review</strong>
235   * 
236   * If your app requests this permission Facebook will have to review how your app uses it.
237   */
238  USER_RELATIONSHIP_DETAILS("user_relationship_details"), //
239
240  /**
241   * Provides access to a person's relationship status, significant other and family members as fields on the User
242   * object.
243   * 
244   * <strong>Review</strong>
245   * 
246   * If your app requests this permission Facebook will have to review how your app uses it.
247   */
248  USER_RELATIONSHIPS("user_relationships"), //
249
250  /**
251   * Provides access to a person's religious and political affiliations.
252   * 
253   * <strong>Review</strong>
254   * 
255   * If your app requests this permission Facebook will have to review how your app uses it.
256   */
257  USER_RELIGION_POLITICS("user_religion_politics"), //
258
259  /**
260   * Provides access to a person's statuses. These are posts on Facebook which don't include links, videos or photos.
261   * 
262   * <strong>Review</strong>
263   * 
264   * If your app requests this permission Facebook will have to review how your app uses it.
265   */
266  USER_STATUS("user_status"), //
267
268  /**
269   * Provides access to the Places a person has been tagged at in photos, videos, statuses and links.
270   * 
271   * <strong>Review</strong>
272   * 
273   * If your app requests this permission Facebook will have to review how your app uses it.
274   */
275  USER_TAGGED_PLACES("user_tagged_places"), //
276
277  /**
278   * Provides access to the videos a person has uploaded or been tagged in.
279   * 
280   * <strong>Review</strong>
281   * 
282   * If your app requests this permission Facebook will have to review how your app uses it.
283   */
284  USER_VIDEOS("user_videos"), //
285
286  /**
287   * Provides access to the person's personal website URL via the website field on the User object.
288   * 
289   * <strong>Review</strong>
290   * 
291   * If your app requests this permission Facebook will have to review how your app uses it.
292   */
293  USER_WEBSITE("user_website"), //
294
295  /**
296   * Provides access to a person's work history and list of employers via the <code>work</code> field on the User
297   * object.
298   * 
299   * <strong>Review</strong>
300   * 
301   * If your app requests this permission Facebook will have to review how your app uses it.
302   */
303  USER_WORK_HISTORY("user_work_history");
304
305  UserDataPermissions(String facebookPermissionString) {
306    this.permissionString = facebookPermissionString;
307  }
308
309  String permissionString;
310
311  @Override
312  public String getPermissionString() {
313    return this.permissionString;
314  }
315
316}
