022package com.restfb.util;
023
024import static com.restfb.util.StringUtils.isBlank;
025
026import java.util.Collection;
027import java.util.Map;
028import java.util.Objects;
029import java.util.function.Supplier;
030
031public class ObjectUtil {
032
033  private ObjectUtil() {
034    // prevent instantiation
035  }
036
037  /**
038   * Ensures that {@code obj} isn't {@code null} or an empty string.
039   *
040   * @param obj
041   *          The parameter to check.
042   * @param errorText
043   *          The exception message.
044   * @throws IllegalArgumentException
045   *           If {@code obj} is {@code null} or an empty string.
046   */
047  public static String requireNotEmpty(String obj, String errorText) {
048    if (isBlank(obj)) {
049      throw new IllegalArgumentException(errorText);
050    }
051    return obj;
052  }
053
054  public static void requireNotEmpty(Collection<?> collection, String errorText) {
055    if (collection == null || collection.isEmpty()) {
056      throw new IllegalArgumentException(errorText);
057    }
058  }
059
060  /**
061   * Ensures that {@code obj} isn't {@code null}.
062   *
063   * @param obj
064   *          The parameter to check.
065   * @param exceptionSupplier
066   *          The supplier for the exception that is thrown if obj is null.
067   * @throws T
068   *           If {@code obj} is {@code null}.
069   */
070  public static <T extends Exception> void requireNotNull(Object obj, Supplier<T> exceptionSupplier) throws T {
071    if (obj == null) {
072      throw exceptionSupplier.get();
073    }
074  }
075
076  /**
077   * Checks is the object is a empty 'collection' or 'map'.
078   * 
079   * @param obj
080   *          the object that is checked
081   * @return {@code true} if the given object is a empty collection or an empty map, {@code false} otherwise
082   */
083  public static boolean isEmptyCollectionOrMap(Object obj) {
084    if (obj instanceof Collection) {
085      return ((Collection) obj).isEmpty();
086    }
087
088    return (obj instanceof Map && ((Map) obj).isEmpty());
089  }
090
091  /**
092   * Ensures that {@code parameter} isn't {@code null} or an empty string.
093   *
094   * @param parameterName
095   *          The name of the parameter (to be used in exception message).
096   * @param parameter
097   *          The parameter to check.
098   * @throws IllegalArgumentException
099   *           If {@code parameter} is {@code null} or an empty string.
100   */
101  public static void verifyParameterPresence(String parameterName, String parameter) {
102    verifyParameterPresence(parameterName, (Object) parameter);
103    requireNotEmpty(parameter, "The '" + parameterName + "' parameter cannot be an empty string.");
104  }
105
106  /**
107   * Ensures that {@code parameter} isn't {@code null}.
108   *
109   * @param parameterName
110   *          The name of the parameter (to be used in exception message).
111   * @param parameter
112   *          The parameter to check.
113   * @throws NullPointerException
114   *           If {@code parameter} is {@code null}.
115   */
116  public static void verifyParameterPresence(String parameterName, Object parameter) {
117    Objects.requireNonNull(parameter, "The '" + parameterName + "' parameter cannot be null.");
118  }
119}
