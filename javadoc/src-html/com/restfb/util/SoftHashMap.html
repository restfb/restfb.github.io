005package com.restfb.util;
006
007import java.lang.ref.*;
008import java.util.*;
009import java.io.Serializable;
010
011public class SoftHashMap<K, V> extends AbstractMap<K, V> implements Serializable {
012  /** The internal HashMap that will hold the SoftReference. */
013  private final Map<K, SoftReference<V>> hash = new HashMap<K, SoftReference<V>>();
014
015  private final Map<SoftReference<V>, K> reverseLookup = new HashMap<SoftReference<V>, K>();
016
017  /** Reference queue for cleared SoftReference objects. */
018  private final ReferenceQueue<V> queue = new ReferenceQueue<V>();
019
020  @Override
021  public V get(Object key) {
022    expungeStaleEntries();
023    V result = null;
024    // We get the SoftReference represented by that key
025    SoftReference<V> soft_ref = hash.get(key);
026    if (soft_ref != null) {
027      // From the SoftReference we get the value, which can be
028      // null if it has been garbage collected
029      result = soft_ref.get();
030      if (result == null) {
031        // If the value has been garbage collected, remove the
032        // entry from the HashMap.
033        hash.remove(key);
034        reverseLookup.remove(soft_ref);
035      }
036    }
037    return result;
038  }
039
040  private void expungeStaleEntries() {
041    Reference<? extends V> sv;
042    while ((sv = queue.poll()) != null) {
043      hash.remove(reverseLookup.remove(sv));
044    }
045  }
046
047  @Override
048  public V put(K key, V value) {
049    expungeStaleEntries();
050    SoftReference<V> soft_ref = new SoftReference<V>(value, queue);
051    reverseLookup.put(soft_ref, key);
052    SoftReference<V> result = hash.put(key, soft_ref);
053    if (result == null) {
054      return null;
055    }
056    reverseLookup.remove(result);
057    return result.get();
058  }
059
060  @Override
061  public V remove(Object key) {
062    expungeStaleEntries();
063    SoftReference<V> result = hash.remove(key);
064    if (result == null) {
065      return null;
066    }
067    return result.get();
068  }
069
070  @Override
071  public void clear() {
072    hash.clear();
073    reverseLookup.clear();
074  }
075
076  @Override
077  public int size() {
078    expungeStaleEntries();
079    return hash.size();
080  }
081
082  /**
083   * Returns a copy of the key/values in the map at the point of calling. However, setValue still sets the value in the
084   * actual SoftHashMap.
085   */
086  @Override
087  public Set<Entry<K, V>> entrySet() {
088    expungeStaleEntries();
089    Set<Entry<K, V>> result = new LinkedHashSet<Entry<K, V>>();
090    for (final Entry<K, SoftReference<V>> entry : hash.entrySet()) {
091      final V value = entry.getValue().get();
092      if (value != null) {
093        result.add(new Entry<K, V>() {
094          @Override
095          public K getKey() {
096            return entry.getKey();
097          }
098
099          @Override
100          public V getValue() {
101            return value;
102          }
103
104          @Override
105          public V setValue(V v) {
106            entry.setValue(new SoftReference<V>(v, queue));
107            return value;
108          }
109        });
110      }
111    }
112    return result;
113  }
114}
